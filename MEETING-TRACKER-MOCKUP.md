async function updateMeetingSchedule() {
  const year = 2026;
  const weekNumber = getCurrentWeekNumber();
  const url = `https://wol.jw.org/en/wol/meetings/r1/lp-e/${year}/${weekNumber}`;

  const meetingData = await scrapeMeetingWorkbook(url);

  await db.meetings.put({
    weekOf: meetingData.weekStartDate,
    midweek: meetingData.midweekMeeting,
    weekend: meetingData.weekendMeeting,
    cachedAt: new Date(),
    year,
    weekNumber
  });

### Notification Timing

**Smart Scheduling:**
- Monday 8 AM: "Midweek meeting Wednesday - prepare!"
- Wednesday 5 PM: "Meeting in 2 hours!" (if meeting at 7 PM)
- Friday 8 AM: "Watchtower study this weekend"
- Saturday 3 PM: "Meeting in 2 hours!" (if meeting at 5 PM)

**Respect User Preferences:**
- Custom times per user
- Quiet hours (e.g., 10 PM - 7 AM)
- Disable during meeting times
- Snooze option

### Offline Strategy

**Pre-cache:**
- Current week: Always available offline
- Next week: Download on Sunday
- Previous 4 weeks: Keep for review
- Watchtower articles: Full text cached
- Videos: Links only (don't cache video files)
